155  public Model LoadGameState()
156  {

158  using (var sr = new StreamReader("GameState.txt"))
159  {
160  this.model = (Model)xs.Deserialize(sr);

163 
164  foreach (Character item in this.model.Player_1.Barricades.Cast<Character>()
165  .Concat(this.model.Player_1.Soldiers.Cast<Character>())

167  .Concat(this.model.Player_2.Soldiers.Cast<Character>())))
168  {
169  if (item.playerID == "Player1")
170  {
171  item.Owner = this.model.Player_1;
172  }
173  else if (item.playerID == "Player2")
174  {
175  item.Owner = this.model.Player_2;
176  }
177  else
178  {
179  throw new Exception();
180  }
181  }
182 
183  List<GameItem> items = this.model.Coins.ConvertAll<GameItem>(x => x);
184  items.AddRange(this.model.Potions.ConvertAll<GameItem>(x => x));
185  items.AddRange(this.model.Player_1.Barricades.ConvertAll<GameItem>(x => x));
186  items.Add(this.model.Player_1.Fortress);
187  items.Add(this.model.Player_1.Selector);
188  items.AddRange(this.model.Player_1.Soldiers.ConvertAll<GameItem>(x => x));
189  items.AddRange(this.model.Player_2.Barricades.ConvertAll<GameItem>(x => x));
190  items.Add(this.model.Player_2.Fortress);
191  items.Add(this.model.Player_2.Selector);
192  items.AddRange(this.model.Player_2.Soldiers.ConvertAll<GameItem>(x => x));
193  foreach (GameItem item in items)

195  item.CX -= Config.FieldWidht;

321  public void MoveSoldier()
322  {
323  foreach (Soldier soldier in this.model.Player_1.Soldiers.Concat(this.model.Player_2.Soldiers))
324  {
325  if (soldier.Enemy != null)
326  {
327  if (soldier.Enemy is Soldier)
328  {
329  (soldier.Enemy as Soldier).Enemy = soldier;
330  }
331 
332  if (this.Attack(soldier.Enemy, soldier.Power))
333  {
334  if (soldier.Enemy is Coin)
335  {
336  soldier.Owner.Money += soldier.Enemy.Bounty;
337  }
338 
339  soldier.Enemy = null;
340  }
341  }
342  else
343  {
344  soldier.CX = (soldier.CX - (Config.FieldWidht / 2)) +
345  (soldier.Owner.Name == this.model.Player_1.Name ? soldier.Speed * Config.StepDistance : soldier.Speed * -1 * Config.StepDistance);
346  foreach (Character item in this.OtherPlayer(soldier).Barricades.Cast<Character>()
347  .Concat(this.OtherPlayer(soldier).Soldiers.Cast<Character>())
348  .Concat(this.model.Coins.Cast<Character>()))
349  {
350  if (soldier.IsCollision(item))
351  {
352  soldier.Enemy = item;
353  }
354  }
355 
356  bool e = false;
357  Potion p = null;
358  foreach (Potion item in this.model.Potions)
359  {
360  if (soldier.IsCollision(item))
361  {
362  this.GetPotion(soldier);
363  e = true;
364  p = item;
365  }
366  }
367 
368  if (e)
369  {
370  this.model.Potions.Remove(p);
371  p = null;
372  }
373 
374  if (soldier.IsCollision(this.OtherPlayer(soldier).Fortress))
375  {
376  soldier.Enemy = this.OtherPlayer(soldier).Fortress;
377  }
378  }
379  }
380 
381  this.IncreasePlayersMoney();
382  this.PlaceExtras();
383  this.RefreshScreen?.Invoke(this, EventArgs.Empty);

412  break;
413  case Characters.Rider:
414  if (player.Money - player.RiderPrice < 0)
415  {
416  break;
417  }
418 
419  player.Money -= player.RiderPrice;
420  player.Soldiers.Add(
421  new Rider(player)

423  CX = player.Name == this.model.Player_1.Name ?
424  (-Config.FieldWidht / 2) - Config.CharacterTileWidth : Config.FieldWidht / 2,
425  Y_Tile = y,

427  break;
428  case Characters.Barricade:
429  if (player.Money - player.BarricadePrice < 0)

433

436  int tX = -Config.FieldWidht / 2;

443  {
444  player.Money -= player.BarricadePrice;
445  player.Barricades.Add(
446  new Barricade(player)

448  CX = tX,
449  Y_Tile = y,
450  });
451  }
452  }
453  else

458  tX1 -= Config.CharacterTileWidth;
459  }
460

463  player.Money -= player.BarricadePrice;

465  new Barricade(player)

467  CX = tX1,

473  break;
