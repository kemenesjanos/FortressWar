102  [Test]
103  public void TestMoveSoldier()
104  {
105  this.logic.StartGame();
106  this.model.Player_1.Soldiers.Clear();
107  this.model.Player_2.Soldiers.Clear();
108 
109  // Knight vs knight same line.
110  this.logic.NewCharacter(Characters.Knight, this.model.Player_1, 1);
111  this.logic.NewCharacter(Characters.Knight, this.model.Player_2, 1);
112  int z = 0;
113 
114  this.model.Player_1.Soldiers.FirstOrDefault().CX += 10 * Config.StepDistance *
115  this.model.Player_1.Soldiers.FirstOrDefault().Speed;
116  this.model.Player_2.Soldiers.FirstOrDefault().CX -= 10 * Config.StepDistance *
117  this.model.Player_1.Soldiers.FirstOrDefault().Speed;
118 
119  while (!this.model.Player_1.Soldiers.FirstOrDefault().IsCollision(
120  this.model.Player_2.Soldiers.FirstOrDefault()) && z < 500)
121  {
122  this.logic.MoveSoldier();
123  z++;
124  }
125 
126  Assert.That(z < 500);
127 
128  // Attack is good.
129  this.logic.MoveSoldier();
130  Assert.That(this.model.Player_1.Soldiers.FirstOrDefault().Life < Config.KnightBaseLife ||
131  this.model.Player_2.Soldiers.FirstOrDefault().Life < Config.KnightBaseLife);
132 
133  // Attack until die
134  z = 0;
135  while (this.model.Player_1.Soldiers.FirstOrDefault() != null &&
136  this.model.Player_2.Soldiers.FirstOrDefault() != null && z < 100)
137  {
138  this.logic.MoveSoldier();
139  z++;
140  }
141 
142  Assert.That(z < 100);
143 
144  // Different lines
145  this.model.Player_1.Soldiers.Clear();
146  this.model.Player_2.Soldiers.Clear();
147 
148  this.logic.NewCharacter(Characters.Knight, this.model.Player_1, 1);
149  this.logic.NewCharacter(Characters.Knight, this.model.Player_2, 2);
150  z = 0;
151 
152  this.model.Player_1.Soldiers.FirstOrDefault().CX += 70 * Config.StepDistance *
153  this.model.Player_1.Soldiers.FirstOrDefault().Speed;
154  this.model.Player_2.Soldiers.FirstOrDefault().CX -= 70 * Config.StepDistance *
155  this.model.Player_1.Soldiers.FirstOrDefault().Speed;
156 
157  while (!this.model.Player_1.Soldiers.FirstOrDefault().IsCollision(
158  this.model.Player_2.Soldiers.FirstOrDefault()) && z < 50)
159  {
160  this.logic.MoveSoldier();
161  z++;
162  }
163 
164  Assert.That(z == 50);
165 
166  // Barricade vs rider
167  this.model.Player_1.Soldiers.Clear();
168  this.model.Player_2.Soldiers.Clear();
169 
170  this.logic.NewCharacter(Characters.Rider, this.model.Player_1, 1);
171  this.logic.NewCharacter(Characters.Barricade, this.model.Player_2, 1);
172  z = 0;
173 
174  Assert.IsNotNull(this.model.Player_2.Barricades.FirstOrDefault());
175  while (this.model.Player_2.Barricades.FirstOrDefault() != null && z < 500)
176  {
177  this.logic.MoveSoldier();
178  z++;
179  }
180 
181  Assert.That(z < 500);
